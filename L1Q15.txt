float* sort(float *arr, float  size, float biteSize) {
    int k;
    for(int i = 0; i < size; i++) {
        for(int j = i + 1; j < size; j++) {
            if(arr[j] < arr[i]) {
                k = arr[i];
                arr[i] = arr[j];
                arr[j] = k;
            }
        }
    }

    return arr;
}
